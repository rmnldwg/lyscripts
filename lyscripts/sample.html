lyscripts.sample

Learn the spread probabilities of the HMM for lymphatic tumor progression using the preprocessed data as input and MCMC as sampling method.

This is the central script performing for our project on modelling lymphatic spread in head & neck cancer. We use it for model comparison via the thermodynamic integration functionality and use the sampled parameter estimates for risk predictions. This risk estimate may in turn some day guide clinicians to make more objective decisions with respect to defining the elective clinical target volume (CTV-N) in radiotherapy.

106class ConvenienceSampler(emcee.EnsembleSampler):
107    """Class that adds some useful defaults to the `emcee.EnsembleSampler`."""
108    def __init__(
109        self,
110        nwalkers,
111        ndim,
112        log_prob_fn,
113        pool=None,
114        backend=None,
115    ):
116        """Initialize sampler with sane defaults."""
117        moves = [(emcee.moves.DEMove(), 0.8), (emcee.moves.DESnookerMove(), 0.2)]
118        super().__init__(nwalkers, ndim, log_prob_fn, pool, moves, backend=backend)
119
120    def run_sampling(
121        self,
122        min_steps: int = 0,
123        max_steps: int = 10000,
124        thin_by: int = 1,
125        initial_state: Optional[Union[emcee.State, np.ndarray]] = None,
126        check_interval: int = 100,
127        trust_threshold: float = 50.,
128        rel_acor_threshold: float = 0.05,
129        progress_desc: Optional[str] = None,
130        **kwargs,
131    ) -> Dict[str, Any]:
132        """Run a round of sampling with at least `min_steps`, at most `max_steps` and
133        only keep every `thin_by` step.
134
135        Every `check_interval` iterations, convergence is checked based on
136        three criteria:
137        - did it run for at least `min_steps`?
138        - has the acor time crossed the N / `trust_threshold`?
139        - did the acor time change less than `rel_acor_threshold`?
140
141        If a `progress_desc` is provided, progress will be displayed.
142
143        Returns dictionary containing the autocorrelation times obtained during the
144        run (along with the iteration number at which they were computed) and the
145        numpy array with the coordinates of the last draw.
146        """
147        if max_steps < min_steps:
148            warnings.warn(
149                "Sampling param min_steps is larger than max_steps. Swapping."
150            )
151            tmp = max_steps
152            max_steps = min_steps
153            min_steps = tmp
154
155        if initial_state is None:
156            initial_state = np.random.uniform(
157                low=0., high=1.,
158                size=(self.nwalkers, self.ndim)
159            )
160
161        iterations = []
162        acor_times = []
163        accept_rates = []
164        old_acor = np.inf
165        is_converged = False
166
167        samples_iterator = self.sample(
168            initial_state=initial_state,
169            iterations=max_steps,
170            thin_by=thin_by,
171            **kwargs
172        )
173        # wrap the iteration over samples in a rich tracker,
174        # if verbosity is desired
175        if progress_desc is not None:
176            report_progress = CustomProgress(console=report)
177            samples_iterator = report_progress.track(
178                sequence=samples_iterator,
179                total=max_steps,
180                description=progress_desc,
181            )
182            report_progress.start()
183
184        coords = None
185        for sample in samples_iterator:
186            # after `check_interval` number of samples...
187            coords = sample.coords
188            if self.iteration < min_steps or self.iteration % check_interval:
189                continue
190
191            # ...get the acceptance rate up to this point
192            accept_rates.append(100. * np.mean(self.acceptance_fraction))
193
194            # ...compute the autocorrelation time and store it in an array.
195            new_acor = self.get_autocorr_time(tol=0)
196            iterations.append(self.iteration)
197            acor_times.append(np.mean(new_acor))
198
199            # check convergence
200            is_converged = self.iteration >= min_steps
201            is_converged &= np.all(new_acor * trust_threshold < self.iteration)
202            rel_acor_diff = np.abs(old_acor - new_acor) / new_acor
203            is_converged &= np.all(rel_acor_diff < rel_acor_threshold)
204
205            # if it has converged, stop
206            if is_converged:
207                break
208
209            old_acor = new_acor
210
211        iterations.append(self.iteration)
212        acor_times.append(np.mean(self.get_autocorr_time(tol=0)))
213        accept_rates.append(100. * np.mean(self.acceptance_fraction))
214        accept_rate_str = f"acceptance rate was {accept_rates[-1]:.2f}%"
215
216        if progress_desc is not None:
217            report_progress.stop()
218            if is_converged:
219                report.success(
220                    progress_desc, "converged,", accept_rate_str
221                )
222            else:
223                report.info(
224                    progress_desc, "finished: Max. steps reached,", accept_rate_str
225                )
226
227        return {
228            "iterations": iterations,
229            "acor_times": acor_times,
230            "accept_rates": accept_rates,
231            "final_state": coords,
232        }

234def run_mcmc_with_burnin(
235    nwalkers: int,
236    ndim: int,
237    log_prob_fn: Callable,
238    nsteps: int,
239    persistent_backend: emcee.backends.HDFBackend,
240    sampling_kwargs: Optional[dict] = None,
241    burnin: Optional[int] = None,
242    keep_burnin: bool = False,
243    thin_by: int = 1,
244    npools: Optional[int] = None,
245    verbose: bool = True,
246) -> Dict[str, Any]:
247    """
248    Draw samples from the `log_prob_fn` using the `ConvenienceSampler` (subclass of
249    `emcee.EnsembleSampler`).
250
251    This function first draws `burnin` samples (that are discarded if `keep_burnin`
252    is set to `False`) and afterwards it performs a second sampling round, starting
253    where the burnin left off, of length `nsteps` that will be stored in the
254    `persistent_backend`.
255
256    When `burnin` is not given, the burnin phase will still take place and it will
257    sample until convergence using convergence criteria defined via `sampling_kwargs`,
258    after which it will draw another `nsteps` samples.
259
260    Only every `thin_by` step will be made persistent.
261
262    If `verbose` is set to `True`, the progress will be displayed.
263
264    Returns a dictionary with some information about the burnin phase.
265    """
266    if sampling_kwargs is None:
267        sampling_kwargs = {}
268
269    if npools is None:
270        created_pool = Pool(os.cpu_count())
271    elif npools == 0:
272        created_pool = DummyPool()
273    elif 0 < npools:
274        created_pool = Pool(np.minimum(npools, os.cpu_count()))
275    else:
276        raise ValueError(
277            "Number of pools must be integer larger or equal to 0 (or `None`)"
278        )
279
280    with created_pool as pool:
281        # Burnin phase
282        if keep_burnin:
283            burnin_backend = persistent_backend
284        else:
285            burnin_backend = emcee.backends.Backend()
286
287        burnin_sampler = ConvenienceSampler(
288            nwalkers, ndim, log_prob_fn,
289            pool=pool, backend=burnin_backend,
290        )
291
292        if burnin is None:
293            burnin_result = burnin_sampler.run_sampling(
294                progress_desc=f"Burn-in ({created_pool._processes} cores)" if verbose else None,
295                **sampling_kwargs,
296            )
297        else:
298            burnin_result = burnin_sampler.run_sampling(
299                min_steps=burnin,
300                max_steps=burnin,
301                progress_desc=f"Burn-in ({created_pool._processes} cores)" if verbose else None,
302            )
303
304        # persistent sampling phase
305        sampler = ConvenienceSampler(
306            nwalkers, ndim, log_prob_fn,
307            pool=pool, backend=persistent_backend,
308        )
309        sampler.run_sampling(
310            min_steps=nsteps,
311            max_steps=nsteps,
312            thin_by=thin_by,
313            initial_state=burnin_result["final_state"],
314            progress_desc=f"Sampling ({created_pool._processes} cores)" if verbose else None,
315        )
316
317        return burnin_result

323def log_prob_fn(theta: np.array) -> float:
324    """log probability function using global variables because of pickling."""
325    llh = MODEL.likelihood(given_params=theta)
326    if np.isinf(llh):
327        return -np.inf, -np.inf
328    return INV_TEMP * llh, llh

331def main(args: argparse.Namespace):
332    """
333    First, this program reads in the parameter YAML file and the CSV training data.
334    After that, it parses the loaded configuration and creates an instance of the model
335    that loads the training data.
336
337    From there - depending on the user input - it either performs thermodynamic
338    integration [^1] to compute the data's evidence given this particular model or it
339    draws directly from the posterior of the parameters given the data. In the latter
340    case it automatically samples until convergence. It uses the amazing
341    [`emcee`](https://github.com/dfm/emcee) library for the MCMC process.
342
343    The drawn samples are always stored in and HDF5 file and some progress about the
344    sampling procedures is stored as well (acceptance rates of samples, estimates
345    of the chains' autocorrelation times and the accuracy during the runs).
346
347    Th help via `lyscripts sample --help` shows this output:
348
349    ```
350    usage: lyscripts sample [-h] [--params PARAMS]
351                            [--modalities MODALITIES [MODALITIES ...]] [--plots PLOTS]
352                            [--ti] [--pools POOLS]
353                            input output
354
355    Learn the spread probabilities of the HMM for lymphatic tumor progression using the
356    preprocessed data as input and MCMC as sampling method.
357
358    This is the central script performing for our project on modelling lymphatic spread
359    in head & neck cancer. We use it for model comparison via the thermodynamic
360    integration functionality and use the sampled parameter estimates for risk
361    predictions. This risk estimate may in turn some day guide clinicians to make more
362    objective decisions with respect to defining the *elective clinical target volume*
363    (CTV-N) in radiotherapy.
364
365
366    POSITIONAL ARGUMENTS
367    input                                 Path to training data files
368    output                                Path to the HDF5 file to store the results in
369
370    OPTIONAL ARGUMENTS
371    -h, --help                            show this help message and exit
372    --params PARAMS                       Path to parameter file (default:
373                                          ./params.yaml)
374    --modalities MODALITIES [MODALITIES ...]
375                                          List of modalities for inference. Must be
376                                          defined in `params.yaml` (default:
377                                          ['max_llh'])
378    --plots PLOTS                         Directory to store plot of acor times
379                                          (default: ./plots)
380    --ti                                  Perform thermodynamic integration (default:
381                                          False)
382    --pools POOLS                         Number of parallel worker pools (CPU cores)
383                                          to use. If not provided, it will use all
384                                          cores. If set to zero, multiprocessing will
385                                          not be used. (default: None)
386    ```
387
388    [^1]: https://doi.org/10.1007/s11571-021-09696-9
389    """
390    with report.status("Read in parameters..."):
391        with open(args.params, mode='r') as params_file:
392            params = yaml.safe_load(params_file)
393        report.success(f"Read in params from {args.params}")
394
395    with report.status("Read in training data..."):
396        # Only read in two header rows when using the Unilateral model
397        is_unilateral = params["model"]["class"] == "Unilateral"
398        header = [0, 1] if is_unilateral else [0, 1, 2]
399        inference_data = pd.read_csv(args.input, header=header)
400        report.success(f"Read in training data from {args.input}")
401
402    with report.status("Set up model & load data..."):
403        global MODEL  # ugly, but necessary for pickling
404        MODEL = model_from_config(
405            graph_params=params["graph"],
406            model_params=params["model"],
407            modalities_params=get_modalities_subset(
408                defined_modalities=params["modalities"],
409                selection=args.modalities,
410            ),
411        )
412        MODEL.patient_data = inference_data
413        ndim = len(MODEL.spread_probs) + MODEL.diag_time_dists.num_parametric
414        nwalkers = ndim * params["sampling"]["walkers_per_dim"]
415        thin_by = params["sampling"]["thin_by"]
416        report.success(
417            f"Set up {type(MODEL)} model with {ndim} parameters and loaded "
418            f"{len(inference_data)} patients"
419        )
420
421    if args.ti:
422        global INV_TEMP
423        with report.status("Prepare thermodynamic integration..."):
424            # make sure path to output file exists
425            args.output.parent.mkdir(parents=True, exist_ok=True)
426
427            # set up sampling params
428            temp_schedule = params["sampling"]["temp_schedule"]
429            nsteps = params["sampling"]["nsteps"]
430            burnin = params["sampling"]["burnin"]
431            report.success("Prepared thermodynamic integration.")
432
433        # record some information about each burnin phase
434        x_axis = temp_schedule.copy()
435        plots = {
436            "acor_times": [],
437            "accept_rates": [],
438        }
439        for i,inv_temp in enumerate(temp_schedule):
440            INV_TEMP = inv_temp
441            report.print(f"TI round {i+1}/{len(temp_schedule)} with β = {INV_TEMP}")
442
443            # set up backend
444            hdf5_backend = emcee.backends.HDFBackend(args.output, name=f"ti/{i+1:0>2d}")
445
446            burnin_info = run_mcmc_with_burnin(
447                nwalkers, ndim, log_prob_fn, nsteps,
448                persistent_backend=hdf5_backend,
449                burnin=burnin,
450                keep_burnin=False,
451                thin_by=thin_by,
452                verbose=True,
453                npools=args.pools,
454            )
455            plots["acor_times"].append(burnin_info["acor_times"][-1])
456            plots["accept_rates"].append(burnin_info["accept_rates"][-1])
457
458        # copy last sampling round over to a group in the HDF5 file called "mcmc"
459        # because that is what other scripts expect to see, e.g. for plotting risks
460        h5_file = h5py.File(args.output, "r+")
461        h5_file.copy(f"ti/{len(temp_schedule):0>2d}", h5_file, name="mcmc")
462        report.success("Finished thermodynamic integration.")
463
464    else:
465        with report.status("Prepare sampling params & backend..."):
466            # make sure path to output file exists
467            args.output.parent.mkdir(parents=True, exist_ok=True)
468
469            # prepare backend
470            hdf5_backend = emcee.backends.HDFBackend(args.output, name="mcmc")
471
472            report.success(f"Prepared sampling params & backend at {args.output}")
473
474        burnin_info = run_mcmc_with_burnin(
475            nwalkers, ndim, log_prob_fn,
476            nsteps=params["sampling"]["nsteps"],
477            persistent_backend=hdf5_backend,
478            sampling_kwargs=params["sampling"]["kwargs"],
479            thin_by=thin_by,
480            verbose=True,
481            npools=args.pools,
482        )
483        x_axis = np.array(burnin_info["iterations"])
484        plots = {
485            "acor_times": burnin_info["acor_times"],
486            "accept_rates": burnin_info["accept_rates"]
487        }
488
489    with report.status("Store plots about burnin phases..."):
490        args.plots.mkdir(parents=True, exist_ok=True)
491
492        for name, y_axis in plots.items():
493            tmp_df = pd.DataFrame(
494                np.array([x_axis, y_axis]).T,
495                columns=["x", name],
496            )
497            tmp_df.to_csv(args.plots/(name + ".csv"), index=False)
498
499        report.success(
500            f"Stored {len(plots)} plots about burnin phases at {args.plots}"
501        )
502
503
504if __name__ == "__main__":
505    parser = argparse.ArgumentParser(description=__doc__)
506    _add_arguments(parser)
507
508    args = parser.parse_args()
509    args.run_main(args)

121    def run_sampling(
122        self,
123        min_steps: int = 0,
124        max_steps: int = 10000,
125        thin_by: int = 1,
126        initial_state: Optional[Union[emcee.State, np.ndarray]] = None,
127        check_interval: int = 100,
128        trust_threshold: float = 50.,
129        rel_acor_threshold: float = 0.05,
130        progress_desc: Optional[str] = None,
131        **kwargs,
132    ) -> Dict[str, Any]:
133        """Run a round of sampling with at least `min_steps`, at most `max_steps` and
134        only keep every `thin_by` step.
135
136        Every `check_interval` iterations, convergence is checked based on
137        three criteria:
138        - did it run for at least `min_steps`?
139        - has the acor time crossed the N / `trust_threshold`?
140        - did the acor time change less than `rel_acor_threshold`?
141
142        If a `progress_desc` is provided, progress will be displayed.
143
144        Returns dictionary containing the autocorrelation times obtained during the
145        run (along with the iteration number at which they were computed) and the
146        numpy array with the coordinates of the last draw.
147        """
148        if max_steps < min_steps:
149            warnings.warn(
150                "Sampling param min_steps is larger than max_steps. Swapping."
151            )
152            tmp = max_steps
153            max_steps = min_steps
154            min_steps = tmp
155
156        if initial_state is None:
157            initial_state = np.random.uniform(
158                low=0., high=1.,
159                size=(self.nwalkers, self.ndim)
160            )
161
162        iterations = []
163        acor_times = []
164        accept_rates = []
165        old_acor = np.inf
166        is_converged = False
167
168        samples_iterator = self.sample(
169            initial_state=initial_state,
170            iterations=max_steps,
171            thin_by=thin_by,
172            **kwargs
173        )
174        # wrap the iteration over samples in a rich tracker,
175        # if verbosity is desired
176        if progress_desc is not None:
177            report_progress = CustomProgress(console=report)
178            samples_iterator = report_progress.track(
179                sequence=samples_iterator,
180                total=max_steps,
181                description=progress_desc,
182            )
183            report_progress.start()
184
185        coords = None
186        for sample in samples_iterator:
187            # after `check_interval` number of samples...
188            coords = sample.coords
189            if self.iteration < min_steps or self.iteration % check_interval:
190                continue
191
192            # ...get the acceptance rate up to this point
193            accept_rates.append(100. * np.mean(self.acceptance_fraction))
194
195            # ...compute the autocorrelation time and store it in an array.
196            new_acor = self.get_autocorr_time(tol=0)
197            iterations.append(self.iteration)
198            acor_times.append(np.mean(new_acor))
199
200            # check convergence
201            is_converged = self.iteration >= min_steps
202            is_converged &= np.all(new_acor * trust_threshold < self.iteration)
203            rel_acor_diff = np.abs(old_acor - new_acor) / new_acor
204            is_converged &= np.all(rel_acor_diff < rel_acor_threshold)
205
206            # if it has converged, stop
207            if is_converged:
208                break
209
210            old_acor = new_acor
211
212        iterations.append(self.iteration)
213        acor_times.append(np.mean(self.get_autocorr_time(tol=0)))
214        accept_rates.append(100. * np.mean(self.acceptance_fraction))
215        accept_rate_str = f"acceptance rate was {accept_rates[-1]:.2f}%"
216
217        if progress_desc is not None:
218            report_progress.stop()
219            if is_converged:
220                report.success(
221                    progress_desc, "converged,", accept_rate_str
222                )
223            else:
224                report.info(
225                    progress_desc, "finished: Max. steps reached,", accept_rate_str
226                )
227
228        return {
229            "iterations": iterations,
230            "acor_times": acor_times,
231            "accept_rates": accept_rates,
232            "final_state": coords,
233        }
