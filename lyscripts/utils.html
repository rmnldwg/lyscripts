lyscripts.utils

This module contains frequently used functions and decorators that are used throughout the subcommands to load e.g. YAML specifications or model definitions.

It also contains helpers for reporting the script's progress via a slightly customized rich console and a custom Exception called LyScriptsWarning that can propagate occuring issues to the right place.

 57        """Extract the `level` of the message (can be "info", "warning" or "error")."""

 74def redirect_to_streamlit(func: Callable) -> Callable:
 75    """
 76    If this method detects that it is called from within a `streamlit`
 77    application, the output is redirected to the appropriate `streamlit` function.
 78    """
 79    func_name = func.__name__
 80
 81    def inner(self, *objects, **kwargs) -> Any:
 82        """Wrapper function."""
 83        if is_streamlit_running():
 84            return getattr(streamlit, func_name)(" ".join(objects))
 85
 86        return func(self, *objects, **kwargs)
 87
 88    return inner

 91def inject_lvl_and_symbol(objects, level = "INFO", symbol = None, width = 8):
 92    """Nicely format the `objects` to be printed with a `level` and `symbol`."""
 93    prefix = "[blue]" + level.ljust(width) + "[/blue]"
 94    if symbol is not None:
 95        objects = [prefix, symbol, *objects]
 96    else:
 97        objects = [prefix, *objects]
 98    return objects

163def binom_pmf(k: list[int] | np.ndarray, n: int, p: float):
164    """Binomial PMF"""
165    if p > 1. or p < 0.:
166        raise ValueError("Binomial prob must be btw. 0 and 1")
167    q = 1. - p
168    binom_coeff = factorial(n) / (factorial(k) * factorial(n - k))
169    return binom_coeff * p**k * q**(n - k)
170

172def parametric_binom_pmf(support: np.ndarray, p: float = 0.5) -> Callable:
173    """Return a parametric binomial PMF"""
174    return binom_pmf(k=support, n=support[-1], p=p)

177def graph_from_config(graph_params: dict):
178    """Build graph dictionary for the `lymph` models from the YAML params."""
179    lymph_graph = {}
180
181    if not "tumor" in graph_params and "lnl" in graph_params:
182        raise KeyError("Parameters must define tumors and LNLs")
183
184    for node_type, node_dict in graph_params.items():
185        for node_name, out_connections in node_dict.items():
186            lymph_graph[(node_type, node_name)] = out_connections
187
188    return lymph_graph

191def add_tstage_marg(
192    model: LymphModel,
193    t_stages: list[str],
194    first_binom_prob: float,
195    max_t: int,
196):
197    """Add margializors over diagnose times to `model`."""
198    for i,stage in enumerate(t_stages):
199        if i == 0:
200            model.diag_time_dists[stage] = binom_pmf(
201                k=np.arange(max_t + 1),
202                n=max_t,
203                p=first_binom_prob
204            )
205        else:
206            model.diag_time_dists[stage] = parametric_binom_pmf

209def model_from_config(
210    graph_params: dict[str, Any],
211    model_params: dict[str, Any],
212    modalities_params: dict[str, Any] | None = None,
213) -> LymphModel:
214    """Create a model instance as defined by some YAML params."""
215    graph = graph_from_config(graph_params)

218    model = model_cls(graph, **model_params["kwargs"])

220    if modalities_params is not None:
221        model.modalities = modalities_params
222
223    add_tstage_marg(
224        model,
225        t_stages=model_params["t_stages"],
226        first_binom_prob=model_params["first_binom_prob"],
227        max_t=model_params["max_t"],
228    )
229
230    return model

233@log_state()
234def create_model_from_config(params: dict[str, Any]) -> LymphModel:
235    """Create a model instance as defined by some YAML params."""
236    if "graph" in params:
237        graph = graph_from_config(params["graph"])
238    else:
239        raise LyScriptsWarning("No graph definition found in YAML file", level="error")
240
241    if "model" in params:
242        model_cls = getattr(models, params["model"]["class"])
243        if not "is_symmetric" in params["model"]["kwargs"]:
244            warnings.warn(
245                "The keywords `base_symmetric`, `trans_symmetric`, and `use_mixing` "
246                "have been deprecated. Please use `is_symmetric` instead.",
247                DeprecationWarning,
248            )
249            params["model"]["kwargs"]["is_symmetric"] = {
250                "tumor_spread": params["model"]["kwargs"].pop("base_symmetric", False),
251                "lnl_spread": params["model"]["kwargs"].pop("trans_symmetric", True),
252            }
253        model = model_cls(graph, **params["model"]["kwargs"])
254
255        add_tstage_marg(
256            model,
257            t_stages=params["model"]["t_stages"],
258            first_binom_prob=params["model"]["first_binom_prob"],
259            max_t=params["model"]["max_t"],
260        )
261    else:
262        raise LyScriptsWarning(
263            "No model class and constructor params found in YAML file",
264            level="error",
265        )
266
267    if "modalities" in params:
268        model.modalities = params["modalities"]
269
270    return model

273def get_dict_depth(nested: dict) -> int:
274    """
275    Get the depth of a nested dictionary.
276
277    For example:
278    >>> get_dict_depth({"a": {"b": 1}})
279    2
280    >>> varying_depth = {"a": {"b": 1}, "c": {"d": {"e": 2}}}
281    >>> get_dict_depth(varying_depth)
282    3
283    """
284    if isinstance(nested, dict):
285        max_depth = None
286        for _, value in nested.items():
287            value_depth = get_dict_depth(value)
288            max_depth = max(max_depth or value_depth, value_depth)
289
290        return 1 + (max_depth or 0)
291
292    return 0

295def delete_private_keys(nested: dict) -> dict:
296    """
297    Delete private keys from a nested dictionary.
298
299    A 'private' key is a key whose name starts with an underscore. For example:
300    >>> delete_private_keys({"patient": {"__doc__": "some patient info", "age": 61}})
301    {'patient': {'age': 61}}
302    >>> delete_private_keys({"patient": {"age": 61}})
303    {'patient': {'age': 61}}
304    """
305    cleaned = {}
306
307    if isinstance(nested, dict):
308        for key, value in nested.items():
309            if not (isinstance(key, str) and key.startswith("_")):
310                cleaned[key] = delete_private_keys(value)
311    else:
312        cleaned = nested
313
314    return cleaned

317def flatten(
318    nested: dict,
319    prev_key: tuple = (),
320    max_depth: int | None = None,
321) -> dict:
322    """
323    Flatten a `nested` dictionary by creating key tuples for each value at `max_depth`.
324
325    For example:
326    >>> nested = {"tumor": {"1": {"t_stage": 1, "size": 12.3}}}
327    >>> flatten(nested)
328    {('tumor', '1', 't_stage'): 1, ('tumor', '1', 'size'): 12.3}
329    >>> mapping = {"patient": {"#": {"age": {"func": int, "columns": ["age"]}}}}
330    >>> flatten(mapping, max_depth=3)
331    {('patient', '#', 'age'): {'func': <class 'int'>, 'columns': ['age']}}
332
333    Note that flattening an already flat dictionary will yield some weird results.
334    """
335    result = {}
336
337    for key, value in nested.items():
338        is_dict = isinstance(value, dict)
339        has_reached_max_depth = max_depth is not None and len(prev_key) >= max_depth - 1
340
341        if is_dict and not has_reached_max_depth:
342            result.update(flatten(value, (*prev_key, key), max_depth))
343        else:
344            result[(*prev_key, key)] = value
345
346    return result

349def unflatten(flat: dict) -> dict:
350    """
351    Take a flat dictionary with tuples of keys and create nested dict from it.
352
353    Like so:
354    >>> flat = {('tumor', '1', 't_stage'): 1, ('tumor', '1', 'size'): 12.3}
355    >>> unflatten(flat)
356    {'tumor': {'1': {'t_stage': 1, 'size': 12.3}}}
357    >>> mapping = {('patient', '#', 'age'): {'func': int, 'columns': ['age']}}
358    >>> unflatten(mapping)
359    {'patient': {'#': {'age': {'func': <class 'int'>, 'columns': ['age']}}}}
360    """
361    result = {}
362
363    for keys, value in flat.items():
364        current = result
365        for key in keys[:-1]:
366            current = current.setdefault(key, {})
367
368        current[keys[-1]] = value
369
370    return result

373def get_modalities_subset(
374    defined_modalities: dict[str, list[float]],
375    selection: list[str],
376) -> dict[str, list[float]]:
377    """
378    Of the `defined_modalities` return only those mentioned in the `selection`.
379
380    For instance:
381    >>> modalities = {"CT": [0.76, 0.81], "MRI": [0.63, 0.86]}
382    >>> get_modalities_subset(modalities, ["CT"])
383    {'CT': [0.76, 0.81]}
384    """
385    selected_modalities = {}
386    for mod in selection:
387        try:
388            selected_modalities[mod] = defined_modalities[mod]
389        except KeyError as key_err:
390            raise KeyError(f"Modality {mod} has not been defined yet") from key_err
391    return selected_modalities

425@provide_file(is_binary=True)
426def load_hdf5_samples(file: BinaryIO, name: str = "mcmc/chain") -> np.ndarray:
427    """
428    Load a chain of samples from an uploaded HDF5 `file` that is stored in the dataset
429    called `name`.
430    """
431    with h5py.File(file, mode="r") as h5file:
432        try:
433            samples = h5file[name][:]
434        except KeyError as key_err:
435            raise KeyError("Dataset `mcmc` not in the HDF5 file.") from key_err
436
437        new_shape = (samples.shape[0] * samples.shape[1], samples.shape[2])
438        flattened_samples = samples.reshape(new_shape)
439        return flattened_samples

def binom_pmf(k: list[int] | numpy.ndarray, n: int, p: float): 
164def binom_pmf(k: list[int] | np.ndarray, n: int, p: float):
165    """Binomial PMF"""
166    if p > 1. or p < 0.:
167        raise ValueError("Binomial prob must be btw. 0 and 1")
168    q = 1. - p
169    binom_coeff = factorial(n) / (factorial(k) * factorial(n - k))
170    return binom_coeff * p**k * q**(n - k)

def flatten(nested: dict, prev_key: tuple = (), max_depth: int | None = None) -> dict: 
318def flatten(
319    nested: dict,
320    prev_key: tuple = (),
321    max_depth: int | None = None,
322) -> dict:
323    """
324    Flatten a `nested` dictionary by creating key tuples for each value at `max_depth`.
325
326    For example:
327    >>> nested = {"tumor": {"1": {"t_stage": 1, "size": 12.3}}}
328    >>> flatten(nested)
329    {('tumor', '1', 't_stage'): 1, ('tumor', '1', 'size'): 12.3}
330    >>> mapping = {"patient": {"#": {"age": {"func": int, "columns": ["age"]}}}}
331    >>> flatten(mapping, max_depth=3)
332    {('patient', '#', 'age'): {'func': <class 'int'>, 'columns': ['age']}}
333
334    Note that flattening an already flat dictionary will yield some weird results.
335    """
336    result = {}
337
338    for key, value in nested.items():
339        is_dict = isinstance(value, dict)
340        has_reached_max_depth = max_depth is not None and len(prev_key) >= max_depth - 1
341
342        if is_dict and not has_reached_max_depth:
343            result.update(flatten(value, (*prev_key, key), max_depth))
344        else:
345            result[(*prev_key, key)] = value
346
347    return result

350def unflatten(flat: dict) -> dict:
351    """
352    Take a flat dictionary with tuples of keys and create nested dict from it.
353
354    Like so:
355    >>> flat = {('tumor', '1', 't_stage'): 1, ('tumor', '1', 'size'): 12.3}
356    >>> unflatten(flat)
357    {'tumor': {'1': {'t_stage': 1, 'size': 12.3}}}
358    >>> mapping = {('patient', '#', 'age'): {'func': int, 'columns': ['age']}}
359    >>> unflatten(mapping)
360    {'patient': {'#': {'age': {'func': <class 'int'>, 'columns': ['age']}}}}
361    """
362    result = {}
363
364    for keys, value in flat.items():
365        current = result
366        for key in keys[:-1]:
367            current = current.setdefault(key, {})
368
369        current[keys[-1]] = value
370
371    return result
